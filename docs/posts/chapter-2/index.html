One additional thing, base R graphics are kinda ugly. We can use R’s esquisse package to turn them to ggplots.

library(ggplot2)
ugtests %>% 
  ggplot(.) + 
  aes(x=finalgrade) + 
  geom_histogram() +
  labs(x="Final Grade") +
  theme_minimal()

ugtests %>% 
  ggplot(.) + 
  aes(y=Yr3, x=as.character(finalgrade), color=as.character(finalgrade)) + 
  geom_boxplot() +
  labs(x="Final Grade", y="Year 3") +
  theme_minimal() +
  guides(color="none")
